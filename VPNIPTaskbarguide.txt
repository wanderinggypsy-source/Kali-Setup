#!/bin/bash

TUN0_IP=$(ip -4 addr show tun0 2>/dev/null | grep -oP '(?<=inet\s)\d+(\.\d+){3}')
TUN1_IP=$(ip -4 addr show tun1 2>/dev/null | grep -oP '(?<=inet\s)\d+(\.\d+){3}')

if [[ -n "$TUN0_IP" ]]; then
    if [[ -n "$TUN1_IP" ]]; then
        echo "VPN: $TUN0_IP $TUN1_IP"
    else
        echo "VPN: $TUN0_IP"
    fi
else
    echo "VPN: No active VPN"
fi

Argos will now display your VPN IPs in the GNOME top bar and refresh every 5 seconds.

If neither tun0 nor tun1 exists, it will show:

You can change the refresh rate by renaming the file to vpn-ip.10s.sh, vpn-ip.30s.sh, etc.
